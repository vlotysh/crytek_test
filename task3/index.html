CREATE TABLE IF NOT EXISTS `Products` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `Product` varchar(255) NOT NULL,
  `Cost` float NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB  DEFAULT CHARSET=utf8 AUTO_INCREMENT=6 ;

FROM `Products`
WHERE `Cost` < (

    SELECT AVG( Cost )
    FROM `Products`
